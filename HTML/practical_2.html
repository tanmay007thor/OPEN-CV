import numpy as np
import matplotlib.pyplot as plt

def generate_hypermatrix(shape):
    hypermatrix = np.random.randint(0, 255, shape, dtype=np.uint8)
    return hypermatrix

def display_image(image):
    plt.imshow(image, cmap='gray', vmin=0, vmax=255)
    plt.title('Hypermatrix Image')
    plt.colorbar()
    plt.show()

hypermatrix_shape = (30, 40, 3)

hypermatrix_image = generate_hypermatrix(hypermatrix_shape)

display_image(hypermatrix_image)

import numpy as np
import matplotlib.pyplot as plt

def generate_hypermatrix(shape, distribution_params):
    hypermatrix = np.random.normal(loc=distribution_params['mean'], 
                                   scale=distribution_params['std_dev'], 
                                   size=shape)
    
    hypermatrix = np.clip(hypermatrix, 0, 255).astype(np.uint8)
    
    return hypermatrix

def display_image(image):
    plt.imshow(image, cmap='gray', vmin=0, vmax=255)
    plt.title('Hypermatrix Image')
    plt.colorbar()
    plt.show()

hypermatrix_shape =[]
hypermatrix_shape = (30, 40, 3)

distribution_params = {'mean': 127, 'std_dev': 30}

hypermatrix_image = generate_hypermatrix(hypermatrix_shape, {'mean': 127, 'std_dev': 30})


display_image(hypermatrix_image)
